Adicionar no docker-compose quando for subir a app:

services:
  api:
    build:
      context: .
      dockerfile: ./src/Delivery.WebApi/Dockerfile
    container_name: rentalsystem_api
    ports:
      - "5000:80"
    depends_on:
      - postgres
      - rabbitmq
